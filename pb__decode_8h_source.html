15 /* Structure for defining custom input streams. You will need to provide
16  * a callback function to read the bytes from your storage, which can be
17  * for example a file or a network socket.
18  *
19  * The callback must conform to these rules:
20  *
21  * 1) Return false on IO errors. This will cause decoding to abort.
22  * 2) You can use state to store your own data (e.g. buffer pointer),
23  * and rely on pb_read to verify that no-body reads past bytes_left.
24  * 3) Your callback may be used with substreams, in which case bytes_left
25  * is different than from the main stream. Don't use bytes_left to compute
26  * any pointers.
27  */
28 struct pb_istream_s
29 {
30 #ifdef PB_BUFFER_ONLY
31  /* Callback pointer is not used in buffer-only configuration.
32  * Having an int pointer here allows binary compatibility but
33  * gives an error if someone tries to assign callback function.
34  */
35  int *callback;
36 #else
37  bool (*callback)(pb_istream_t *stream, pb_byte_t *buf, size_t count);
38 #endif
39 
40  void *state; /* Free field for use by callback implementation */
41  size_t bytes_left;

55 /* Decode a single protocol buffers message from input stream into a C structure.
56  * Returns true on success, false on any failure.
57  * The actual struct pointed to by dest must match the description in fields.
58  * Callback fields of the destination structure must be initialized by caller.
59  * All other fields will be initialized by this function.
60  *
61  * Example usage:
62  * MyMessage msg = {};
63  * uint8_t buffer[64];
64  * pb_istream_t stream;
65  *
66  * // ... read some data into buffer ...
67  *
68  * stream = pb_istream_from_buffer(buffer, count);
69  * pb_decode(&stream, MyMessage_fields, &msg);
70  */
71 bool pb_decode(pb_istream_t *stream, const pb_field_t fields[], void *dest_struct);

73 /* Same as pb_decode, except does not initialize the destination structure
74  * to default values. This is slightly faster if you need no default values
75  * and just do memset(struct, 0, sizeof(struct)) yourself.
76  *
77  * This can also be used for 'merging' two messages, i.e. update only the
78  * fields that exist in the new message.
79  *
80  * Note: If this function returns with an error, it will not release any
81  * dynamically allocated fields. You will need to call pb_release() yourself.
82  */
83 bool pb_decode_noinit(pb_istream_t *stream, const pb_field_t fields[], void *dest_struct);
