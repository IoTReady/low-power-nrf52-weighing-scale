518 __STATIC_INLINE void nrf_spis_configure(NRF_SPIS_Type * p_reg,
519  nrf_spis_mode_t spi_mode,
520  nrf_spis_bit_order_t spi_bit_order)
521 {
522  uint32_t config = (spi_bit_order == NRF_SPIS_BIT_ORDER_MSB_FIRST ?
523  SPIS_CONFIG_ORDER_MsbFirst : SPIS_CONFIG_ORDER_LsbFirst);
524 
525  switch (spi_mode)
526  {
527  default:
528  case NRF_SPIS_MODE_0:
529  config |= (SPIS_CONFIG_CPOL_ActiveHigh << SPIS_CONFIG_CPOL_Pos) |
530  (SPIS_CONFIG_CPHA_Leading << SPIS_CONFIG_CPHA_Pos);
531  break;
532 
533  case NRF_SPIS_MODE_1:
534  config |= (SPIS_CONFIG_CPOL_ActiveHigh << SPIS_CONFIG_CPOL_Pos) |
535  (SPIS_CONFIG_CPHA_Trailing << SPIS_CONFIG_CPHA_Pos);
536  break;
537 
538  case NRF_SPIS_MODE_2:
539  config |= (SPIS_CONFIG_CPOL_ActiveLow << SPIS_CONFIG_CPOL_Pos) |
540  (SPIS_CONFIG_CPHA_Leading << SPIS_CONFIG_CPHA_Pos);
541  break;
542 
543  case NRF_SPIS_MODE_3:
544  config |= (SPIS_CONFIG_CPOL_ActiveLow << SPIS_CONFIG_CPOL_Pos) |
545  (SPIS_CONFIG_CPHA_Trailing << SPIS_CONFIG_CPHA_Pos);
546  break;
547  }
548  p_reg->CONFIG = config;
549 }
