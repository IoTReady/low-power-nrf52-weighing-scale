672 
673  // Load custom instruction.
674  switch (length)

677  reg |= ((uint32_t)p_tx_data_8[7]) << QSPI_CINSTRDAT1_BYTE7_Pos;
678  /* fall-through */
679  case NRF_QSPI_CINSTR_LEN_8B:
680  reg |= ((uint32_t)p_tx_data_8[6]) << QSPI_CINSTRDAT1_BYTE6_Pos;
681  /* fall-through */
682  case NRF_QSPI_CINSTR_LEN_7B:
683  reg |= ((uint32_t)p_tx_data_8[5]) << QSPI_CINSTRDAT1_BYTE5_Pos;

686  reg |= ((uint32_t)p_tx_data_8[4]);

691  reg |= ((uint32_t)p_tx_data_8[3]) << QSPI_CINSTRDAT0_BYTE3_Pos;
692  /* fall-through */
693  case NRF_QSPI_CINSTR_LEN_4B:
694  reg |= ((uint32_t)p_tx_data_8[2]) << QSPI_CINSTRDAT0_BYTE2_Pos;
695  /* fall-through */
696  case NRF_QSPI_CINSTR_LEN_3B:
697  reg |= ((uint32_t)p_tx_data_8[1]) << QSPI_CINSTRDAT0_BYTE1_Pos;
