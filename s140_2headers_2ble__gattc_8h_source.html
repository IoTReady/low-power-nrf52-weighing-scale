688 __STATIC_INLINE uint32_t sd_ble_gattc_evt_char_val_by_uuid_read_rsp_iter(ble_gattc_evt_t *p_gattc_evt, ble_gattc_handle_value_t *p_iter)
689 {
690  uint32_t value_len = p_gattc_evt->params.char_val_by_uuid_read_rsp.value_len;
691  uint8_t *p_first = p_gattc_evt->params.char_val_by_uuid_read_rsp.handle_value;
692  uint8_t *p_next = p_iter->p_value ? p_iter->p_value + value_len : p_first;
693 
694  if ((p_next - p_first) / (sizeof(uint16_t) + value_len) < p_gattc_evt->params.char_val_by_uuid_read_rsp.count)
695  {
696  p_iter->handle = (uint16_t)p_next[1] << 8 | p_next[0];
697  p_iter->p_value = p_next + sizeof(uint16_t);
698  return NRF_SUCCESS;
699  }
700  else
701  {
702  return NRF_ERROR_NOT_FOUND;
