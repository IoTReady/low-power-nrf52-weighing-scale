465 __STATIC_INLINE bool nrf_i2s_configure(NRF_I2S_Type * p_i2s,
466  nrf_i2s_mode_t mode,
467  nrf_i2s_format_t format,
468  nrf_i2s_align_t alignment,
469  nrf_i2s_swidth_t sample_width,
470  nrf_i2s_channels_t channels,
471  nrf_i2s_mck_t mck_setup,
472  nrf_i2s_ratio_t ratio)
473 {
474  if (mode == NRF_I2S_MODE_MASTER)
475  {
476  // The MCK/LRCK ratio shall be a multiple of 2 * sample width.
477  if (((sample_width == NRF_I2S_SWIDTH_16BIT) &&
478  (ratio == NRF_I2S_RATIO_48X))
479  ||
480  ((sample_width == NRF_I2S_SWIDTH_24BIT) &&
481  ((ratio == NRF_I2S_RATIO_32X) ||
482  (ratio == NRF_I2S_RATIO_64X) ||
483  (ratio == NRF_I2S_RATIO_128X) ||
484  (ratio == NRF_I2S_RATIO_256X) ||
485  (ratio == NRF_I2S_RATIO_512X))))
486  {
487  return false;
488  }
489  }
490 
491  p_i2s->CONFIG.MODE = mode;
492  p_i2s->CONFIG.FORMAT = format;
493  p_i2s->CONFIG.ALIGN = alignment;
494  p_i2s->CONFIG.SWIDTH = sample_width;
495  p_i2s->CONFIG.CHANNELS = channels;
496  p_i2s->CONFIG.RATIO = ratio;
497 
498  if (mck_setup == NRF_I2S_MCK_DISABLED)
499  {
500  p_i2s->CONFIG.MCKEN =
501  (I2S_CONFIG_MCKEN_MCKEN_Disabled << I2S_CONFIG_MCKEN_MCKEN_Pos);
502  }
503  else
504  {
505  p_i2s->CONFIG.MCKFREQ = mck_setup;
506  p_i2s->CONFIG.MCKEN =
507  (I2S_CONFIG_MCKEN_MCKEN_Enabled << I2S_CONFIG_MCKEN_MCKEN_Pos);
508  }
509 
510  return true;
511 }
