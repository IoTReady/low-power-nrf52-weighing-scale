290 __STATIC_INLINE uint32_t __sd_nvic_app_accessible_irq(IRQn_Type IRQn)
291 {
292  if (IRQn < 32)
293  {
294  return ((1UL<<IRQn) & __NRF_NVIC_APP_IRQS_0) != 0;
295  }
296  else if (IRQn < 64)
297  {
298  return ((1UL<<(IRQn-32)) & __NRF_NVIC_APP_IRQS_1) != 0;
299  }
300  else
301  {
302  return 1;
303  }
304 }

324 {

327  return NRF_ERROR_SOC_NVIC_INTERRUPT_NOT_AVAILABLE;

337  }
338  else

340  NVIC_EnableIRQ(IRQn);
341  }

438 __STATIC_INLINE uint32_t sd_nvic_critical_region_enter(uint8_t * p_is_nested_critical_region)
439 {
440  int was_masked = __sd_nvic_irq_disable();
441  if (!nrf_nvic_state.__cr_flag)
442  {
443  nrf_nvic_state.__cr_flag = 1;
444  nrf_nvic_state.__irq_masks[0] = ( NVIC->ICER[0] & __NRF_NVIC_APP_IRQS_0 );
445  NVIC->ICER[0] = __NRF_NVIC_APP_IRQS_0;
446  nrf_nvic_state.__irq_masks[1] = ( NVIC->ICER[1] & __NRF_NVIC_APP_IRQS_1 );
447  NVIC->ICER[1] = __NRF_NVIC_APP_IRQS_1;
448  *p_is_nested_critical_region = 0;
449  }
450  else
451  {
452  *p_is_nested_critical_region = 1;
453  }
454  if (!was_masked)
455  {
456  __sd_nvic_irq_enable();
457  }
458  return NRF_SUCCESS;
459 }

461 __STATIC_INLINE uint32_t sd_nvic_critical_region_exit(uint8_t is_nested_critical_region)
462 {
463  if (nrf_nvic_state.__cr_flag && (is_nested_critical_region == 0))
464  {
465  int was_masked = __sd_nvic_irq_disable();
466  NVIC->ISER[0] = nrf_nvic_state.__irq_masks[0];
467  NVIC->ISER[1] = nrf_nvic_state.__irq_masks[1];
468  nrf_nvic_state.__cr_flag = 0;
469  if (!was_masked)
470  {
471  __sd_nvic_irq_enable();
472  }
473  }
474 
475  return NRF_SUCCESS;
