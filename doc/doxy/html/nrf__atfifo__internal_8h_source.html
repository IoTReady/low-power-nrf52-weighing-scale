166 {
167  /* Registry usage:
168  * R0 - p_fifo
169  * R1 - p_old_tail
170  * R2 - internal variable old_tail (saved by caller)
171  * R3 - internal variable new_tail (saved by caller)
172  * R4 - internal temporary register (saved by this function)
173  * R5 - not used stored to keep the stack aligned to 8 bytes
174  * Returned value:
175  * R0 (bool - 32 bits)
176  */
177  push {r4, r5}
178 nrf_atfifo_wspace_req_repeat
179  /* Load tail tag and set memory monitor !!! R2 - old tail !!! */
180  ldrex r2, [r0, #__cpp(offsetof(nrf_atfifo_t, tail))]
181  /* Extract write position !!! R3 !!! */
182  uxth r3, r2
183  /* Increment address with overload support !!! R4 used temporary !!! */
184  ldrh r4, [r0, #__cpp(offsetof(nrf_atfifo_t, item_size))]
185  add r3, r4
186  ldrh r4, [r0, #__cpp(offsetof(nrf_atfifo_t, buf_size))]
187  cmp r3, r4
188  it hs
189  subhs r3, r3, r4
190 
191  /* Check if FIFO would overload after making this increment !!! R4 used temporary !!! */
192  ldrh r4, [r0, #__cpp(offsetof(nrf_atfifo_t, head) + offsetof(nrf_atfifo_postag_pos_t, wr))]

195  clrexeq

197  beq nrf_atfifo_wspace_req_exit

202
