329 __STATIC_INLINE void nrf_spi_configure(NRF_SPI_Type * p_reg,
330  nrf_spi_mode_t spi_mode,
331  nrf_spi_bit_order_t spi_bit_order)
332 {
333  uint32_t config = (spi_bit_order == NRF_SPI_BIT_ORDER_MSB_FIRST ?
334  SPI_CONFIG_ORDER_MsbFirst : SPI_CONFIG_ORDER_LsbFirst);
335  switch (spi_mode)
336  {
337  default:
338  case NRF_SPI_MODE_0:
339  config |= (SPI_CONFIG_CPOL_ActiveHigh << SPI_CONFIG_CPOL_Pos) |
340  (SPI_CONFIG_CPHA_Leading << SPI_CONFIG_CPHA_Pos);
341  break;
342 
343  case NRF_SPI_MODE_1:
344  config |= (SPI_CONFIG_CPOL_ActiveHigh << SPI_CONFIG_CPOL_Pos) |
345  (SPI_CONFIG_CPHA_Trailing << SPI_CONFIG_CPHA_Pos);
346  break;
347 
348  case NRF_SPI_MODE_2:
349  config |= (SPI_CONFIG_CPOL_ActiveLow << SPI_CONFIG_CPOL_Pos) |
350  (SPI_CONFIG_CPHA_Leading << SPI_CONFIG_CPHA_Pos);
351  break;
352 
353  case NRF_SPI_MODE_3:
354  config |= (SPI_CONFIG_CPOL_ActiveLow << SPI_CONFIG_CPOL_Pos) |
355  (SPI_CONFIG_CPHA_Trailing << SPI_CONFIG_CPHA_Pos);
356  break;
357  }
358  p_reg->CONFIG = config;
359 }
