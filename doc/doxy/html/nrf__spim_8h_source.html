669 __STATIC_INLINE void nrf_spim_configure(NRF_SPIM_Type * p_reg,
670  nrf_spim_mode_t spi_mode,
671  nrf_spim_bit_order_t spi_bit_order)
672 {
673  uint32_t config = (spi_bit_order == NRF_SPIM_BIT_ORDER_MSB_FIRST ?
674  SPIM_CONFIG_ORDER_MsbFirst : SPIM_CONFIG_ORDER_LsbFirst);
675  switch (spi_mode)
676  {
677  default:
678  case NRF_SPIM_MODE_0:
679  config |= (SPIM_CONFIG_CPOL_ActiveHigh << SPIM_CONFIG_CPOL_Pos) |
680  (SPIM_CONFIG_CPHA_Leading << SPIM_CONFIG_CPHA_Pos);
681  break;
682 
683  case NRF_SPIM_MODE_1:
684  config |= (SPIM_CONFIG_CPOL_ActiveHigh << SPIM_CONFIG_CPOL_Pos) |
685  (SPIM_CONFIG_CPHA_Trailing << SPIM_CONFIG_CPHA_Pos);
686  break;
687 
688  case NRF_SPIM_MODE_2:
689  config |= (SPIM_CONFIG_CPOL_ActiveLow << SPIM_CONFIG_CPOL_Pos) |
690  (SPIM_CONFIG_CPHA_Leading << SPIM_CONFIG_CPHA_Pos);
691  break;
692 
693  case NRF_SPIM_MODE_3:
694  config |= (SPIM_CONFIG_CPOL_ActiveLow << SPIM_CONFIG_CPOL_Pos) |
695  (SPIM_CONFIG_CPHA_Trailing << SPIM_CONFIG_CPHA_Pos);
696  break;
697  }
698  p_reg->CONFIG = config;
699 }
