376 __STATIC_INLINE void nrf_gpiote_task_enable(uint32_t idx)
377 {
378  uint32_t final_config = NRF_GPIOTE->CONFIG[idx] | GPIOTE_CONFIG_MODE_Task;
379 #ifdef NRF51
380  /* Workaround for the OUTINIT PAN. When nrf_gpiote_task_config() is called a glitch happens
381  on the GPIO if the GPIO in question is already assigned to GPIOTE and the pin is in the
382  correct state in GPIOTE but not in the OUT register. */
383  /* Configure channel to not existing, not connected to the pin, and configure as a tasks that will set it to proper level */
384  NRF_GPIOTE->CONFIG[idx] = final_config | (((31) << GPIOTE_CONFIG_PSEL_Pos) & GPIOTE_CONFIG_PORT_PIN_Msk);
385  __NOP();
386  __NOP();
387  __NOP();
388 #endif
389  NRF_GPIOTE->CONFIG[idx] = final_config;
390 }
