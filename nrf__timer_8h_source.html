606 __STATIC_INLINE uint32_t nrf_timer_us_to_ticks(uint32_t time_us,
607  nrf_timer_frequency_t frequency)
608 {
609  // The "frequency" parameter here is actually the prescaler value, and the
610  // timer runs at the following frequency: f = 16 MHz / 2^prescaler.
611  uint32_t prescaler = (uint32_t)frequency;
612  NRFX_ASSERT(time_us <= (UINT32_MAX / 16UL));
613  return ((time_us * 16UL) >> prescaler);
614 }
