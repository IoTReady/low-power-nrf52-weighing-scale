191 __STATIC_INLINE void nrf_lpcomp_configure(const nrf_lpcomp_config_t * p_config)
192 {
193  NRF_LPCOMP->TASKS_STOP = 1;
194  NRF_LPCOMP->ENABLE = LPCOMP_ENABLE_ENABLE_Disabled << LPCOMP_ENABLE_ENABLE_Pos;
195  NRF_LPCOMP->REFSEL =
196  (p_config->reference << LPCOMP_REFSEL_REFSEL_Pos) & LPCOMP_REFSEL_REFSEL_Msk;
197 
198  //If external source is choosen extract analog reference index.
199  if ((p_config->reference & LPCOMP_REFSEL_REFSEL_ARef)==LPCOMP_REFSEL_REFSEL_ARef)
200  {
201  uint32_t extref = p_config->reference >> 16;
202  NRF_LPCOMP->EXTREFSEL = (extref << LPCOMP_EXTREFSEL_EXTREFSEL_Pos) & LPCOMP_EXTREFSEL_EXTREFSEL_Msk;
203  }
204 
205  NRF_LPCOMP->ANADETECT =
206  (p_config->detection << LPCOMP_ANADETECT_ANADETECT_Pos) & LPCOMP_ANADETECT_ANADETECT_Msk;
207 #ifdef LPCOMP_FEATURE_HYST_PRESENT
208  NRF_LPCOMP->HYST = ((p_config->hyst) << LPCOMP_HYST_HYST_Pos) & LPCOMP_HYST_HYST_Msk;
209 #endif //LPCOMP_FEATURE_HYST_PRESENT
210  NRF_LPCOMP->SHORTS = 0;
211  NRF_LPCOMP->INTENCLR = LPCOMP_INTENCLR_CROSS_Msk | LPCOMP_INTENCLR_UP_Msk |
212  LPCOMP_INTENCLR_DOWN_Msk | LPCOMP_INTENCLR_READY_Msk;
213 }
